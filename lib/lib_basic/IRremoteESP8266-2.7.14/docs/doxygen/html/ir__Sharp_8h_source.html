37 union SharpProtocol{
38  uint8_t raw[kSharpAcStateLength];
39  struct {
40  // Byte 0~3
41  uint8_t pad[4];
42  // Byte 4
43  uint8_t Temp :4;
44  uint8_t Model :1;
45  uint8_t :3;
46  // Byte 5
47  uint8_t :4;
48  uint8_t PowerSpecial :4;
49  // Byte 6
50  uint8_t Mode :2;
51  uint8_t :1;
52  uint8_t Clean :1;
53  uint8_t Fan :3;
54  uint8_t :1;
55  // Byte 7
56  uint8_t TimerHours :4;
57  uint8_t :2;
58  uint8_t TimerType :1;
59  uint8_t TimerEnabled:1;
60  // Byte 8
61  uint8_t Swing :3;
62  uint8_t :5;
63  // Byte 9
64  uint8_t :8;
65  // Byte 10
66  uint8_t Special :8;
67  // Byte 11
68  uint8_t :2;
69  uint8_t Ion :1;
70  uint8_t :1;
71  uint8_t A705:1;
72  uint8_t :3;
73  // Byte 12
74  uint8_t :4;
75  uint8_t Sum :4;
76  };
