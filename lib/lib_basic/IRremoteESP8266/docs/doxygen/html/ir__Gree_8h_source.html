54  // Byte 3
55  uint8_t :2;
56  uint8_t TempExtraDegreeF:1;
57  uint8_t UseFahrenheit :1;

59  // Byte 4
60  uint8_t Swing:4;
61  uint8_t :0;
62  // Byte 5
63  uint8_t DisplayTemp :2;
64  uint8_t IFeel :1;
65  uint8_t unknown2 :3; // value = 0b100
66  uint8_t WiFi :1;
67  uint8_t :0;
68  // Byte 6
