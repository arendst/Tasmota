38  // only use 48bits
39  struct {

42  // Byte 1
43  uint8_t :8; // value=0xFF
44  // Byte 2

51  uint8_t Mode:3;
52  uint8_t Fan:2;
53  uint8_t :1;
54  uint8_t Sleep:1;

56  // Byte 5
57  uint8_t :0;
58  };
59 };
